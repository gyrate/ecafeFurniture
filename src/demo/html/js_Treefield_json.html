Example:通过url请求获得数据源

 treeField = new Sui.form.TreeField({
    applyTo: 'treeField',
    selectText:true,
    initLabel:'初始值',//指定组件描述值
    value:'408',//指定隐藏表单项的值
    treeConfig:{

        url:'../../data/treefieldData.php',
        requestOnce:true,
        checkedType:Sui.TreeSelectType.SINGLE //节点选择模式
    }
});
