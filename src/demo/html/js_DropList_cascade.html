var select = new Sui.form.DropList({
applyTo : 'input',
name : 'continents',
value:'Asia',
searchable : true,
comboMaxHeight:150,
items : [
    { label : '亚洲', value : 'Asia'},
    { label : '美洲',value : 'Americas'},
    { label : '欧洲',value : 'Europe'},
    { label : '大洋洲',value : 'Oceania'},
    { label : '非洲',value : 'Africa'}
],
listeners:{
    selected : function(event){
        //更新select2的数据源
        select2.setStore(cities[event.value]);
    }
}
});

var items2 = cities[select.getValue()];

var select2 = new Sui.form.DropList({
applyTo:'input2',
name:'city',
items: items2,
listeners:{
    selected: function(event) {
        console.log(event.value);
    }
}
});
select2.setSelectedIndex(0);
