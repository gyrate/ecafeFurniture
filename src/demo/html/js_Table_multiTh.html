Example:创建一个标准Table组件

var store = new Sui.data.Store([
    { taskName : '节点1',  taskType: '编码',parameter:0},
    { taskName : '节点11', taskType: '编码',parameter:0},
    { taskName : '节点111',taskType: '编码',parameter:0 },
    { taskName : '节点112', taskType: '编码',parameter:0},
    { taskName : '节点12',  taskType: '编码',parameter:0},
    { taskName : '节点2', taskType: '编码',parameter:0},
    { taskName : '节点21', taskType: '编码',parameter:0},
    { taskName : '节点22', taskType: '编码',parameter:0}
]);

var defaultRow = {
    taskName : '报工前端开发',
    taskType: '编码'
};

var table = new Sui.table.Table({
    selectType : Sui.table.SelectType.MULTI,
    applyTo : 'grid',
    treeTable : false,
    treeColumnName : 'taskName',
    columns: [
        [
            { display:'all',name:'taskType',width:50,align:'center',rowspan:2 },
            { display:'任务1',name:'miss1',width:60,align:'center',colspan:2 } ,
            { display: '参数值目前没法合并',name:'parameter',width:200, colspan:1,align:'center'}
        ],
        [
            { display: '任务名称', name: 'taskName', width: 60,align:'center' },
            { display: '任务类型', name: 'taskType', width: 100, align: 'center'},
            { display: '参数值目前没法合并',name:'parameter',width:200, colspan:1,align:'center'}
        ]
    ],
    store :  store
});

$("#deleteRow").click(function() {
    table.removeSelectedRows();
});

$("#getSelectedRow").click(function() {
    if(table.getSelectedRowIndexs().length){
        alert('选中第' + table.getSelectedRowIndexs() + '行。');
    }else{
        alert('未选中任何行。');
    }
});

$("#clearSelectedRow").click(function() {
    table.clearSelectedRows();
});

var i = 0;
$("#addRow").click(function() {

    store.addRecordData({
        taskName : "name" + (i++),
        taskType : "sex" + (i++),
        parameter : i
    });

});
