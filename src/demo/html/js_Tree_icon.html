Example:创建一个标准Tree组件

var tree = new Sui.tree.Tree({
    renderTo:'tree',
    lineVisible : true,
    nodeIconVisible : true,

    nodeClasses : {
        country : 'country_node_icon',
        provice : 'provice_node_icon',
        city : 'city_node_icon'
    },

    listeners : {
        nodeClick : function(event) {
            var text = event.nodeText;
            Sui.log(text);
        }
    }
});

var china = new Sui.tree.TreeNode({
    expanded : true,
    nodeText:'中国',
    nodeType : 'country'
});
var hunan = new Sui.tree.TreeNode({
    expanded : true,nodeText:'湖南',
    nodeType : 'provice'
});
var changsha = new Sui.tree.TreeNode({
    expanded : true,nodeText:'长沙',
    nodeType : 'city'
});

var hengyang = new Sui.tree.TreeNode({
    nodeText:'衡阳',
    nodeType : 'city'
});
var hubei = new Sui.tree.TreeNode({
    nodeText:'湖北',
    nodeType : 'provice'
});

china.addChildNode(hunan);
hunan.addChildNode(changsha);
hunan.addChildNode(hengyang);
china.addChildNode(hubei);

tree.setRoot(china);
