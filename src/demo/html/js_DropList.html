    Sui.form.Droplist(${option})

    droplist.js

    类

    无

    参数

    ${option}所能够包含的属性。 *表示该属性为必填项

    参数名(类型) 实例 说明
    *items(Array) [{ label : 'ABBCD', value : '1'},{ label : 'ABBCD2', value : '2' },]

    下拉单选框选项数据

    *applyTo(String) InputID'

    要替换的元素Id

    labelField (String) 'label'(默认) 服务器输出的数据中与描述值对应的属性名，如数据为[{userName:'tom',id:'12',desc:'tom 编号12'}],则可定义labelField为'userName'
    valueField (String) 'value'(默认) 服务器输出的数据中与实际值对应的属性名，如数据为[{userName:'tom',id:'12',desc:'tom 编号12'}],则可定义valueField为'id'
    displayValueField (String) 'displayValue'(默认) 服务器输出的数据中与输入框显示的值对应的属性名，如数据为[{userName:'tom',id:'12',desc:'tom 编号12'}],则可定义displayValueField为'desc'
    name (String) 'user' 提交表单时所对应的属性名
    value (String,Array) '3' 或 ['3','4','5'] 初始化组件时，组件的值
    listeners (Object) {selected:function(event){...}} 监听器，可根据对组件的各种动作（如选中、点击等）执行反应，具体可见演示
    comboMaxHeight (Number) 150 下拉选择框最大高度，默认值为null
    forceSelect (Boolean) true (默认) 是否强制选择选项内的值，非选项内的值不能提交，默认为true
    needIframeInIE (Boolean) false (默认) 在IE浏览器是否需要使用iframe方法将layer的z轴层级提高，高于视频控件 ，该参数只对IE有效
    hideIconElement (Boolean) false (默认) 是否隐藏组件触发选择层按钮
    url (String) null (默认) 通过异步方式获得初始化数据，此时url为请求地址
    searchUrl (String) null (默认) 支持ajax搜索时，数据请求地址，通过GET方式传递关键词返回数据

    例子

    Example:创建一个标准DropList组件

    折叠|展开 
     select1 = new Sui.form.DropList({
    applyTo : 'input',
    name : 'input',
    searchable : true,
    value : '3',
    items : [
        { label : 'ABBABBCDABBCDABBCDABBCDABBCDABBCDCD', value : '1'},
        { label : 'ABBABBCDABBCDABBCDABBCDABBCDABBCDCD2大风吹 大风吹 冰淇淋流泪', value : '2' },
        { label : 'ABBCD3',  value : '3'},
        { label : 'ABBCD4', value : '4'},
        { label : 'DBBCD2',value : '5'},
        { label : 'AAAAA', value : '6'},
        { label : 'BBBBB',value : '7'},
        { label : 'CCCCC', value : '8'},
        { label : 'DDDDD', value : '9'},
        { label : 'EEEEE', value : '10'},
        { label : 'FFFFF', value : '11'},
        { label : 'GGGGG', value : '12'},
        { label : 'HHHHH', value : '13'},
        { label : 'IIIII', value : '14'},
        { label : 'JJJJJ', value : '15'}
    ],
    listeners:{
        selected : function(event){
            alert(event.value);
        }
    }
});

    Example: 设置下拉单选框是否可搜索

    折叠|展开 
    
select1.setSearchable(searchable);
searchable = !searchable;
//首次运行变为不可搜索，再次运行变为可以搜索

    Example: 设置下拉单选框为只读

    折叠|展开 
    
select1.setReadOnly(readonly);
readonly = !readonly;
//首次运行变为只读，再次运行变为可编辑

    Example: 函数清空输入框内容

    折叠|展开 
    
select1.clearValue();

    Example: 强制使用选项的值，不能使用手动输入值

    折叠|展开 
    
select1.forceSelect = !select1.forceSelect;
