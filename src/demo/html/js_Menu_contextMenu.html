Example:创建一个右键菜单组件

var menu = new Sui.menu.ContextMenu({
itemConfigs:[
    {html:'添加节点', action:function(obj) {
        showLog(obj.data.html);
    }},
    {html:'操作B', action:function(obj) {
        showLog(obj.data.html);
    }},
    {html:'操作C', action:function(obj) {
        showLog(obj.data.html);
    }}
]
});

function showLog(msg) {
   alert('你点击了:' + msg);
}

$(document).bind('contextmenu',function(e){
   menu.alignToAndShow(e);
   return false;
})
