Example:创建一个多菜单组件

var menuBar = new Sui.menu.MenuBar({
        cls : 'menubar',
        applyTo : 'toolbar'
    });
    menuBar.addComponent(new Sui.menu.Menu({
        html:'菜单0',
        itemConfigs:[
            { html:'菜单0_1',
                action:function(){
                    alert('in action');
                },
                listeners:{
                    click:function(e){
                        alert('item'+e.text);
                    }
                }
            },
            {html:'菜单0_2'},
            {html:'菜单0_3'}
        ]
    }));
    menuBar.addComponent(new Sui.menu.Menu({
        html:'菜单1',
        itemConfigs:[
            {
                html:'item0'
            },{
                html:'item1'
            },
            {
                html:'item11',
                itemConfigs:[
                    { html:'item11_1'},
                    { html:'item11_2'},
                    { html:'有子菜单',
                        itemConfigs:[
                            {
                                html:'item111',
                                listeners:{ }
                            },{
                                html:'item111',
                                itemConfigs:[
                                    { html:'node1'},
                                    {
                                        html:'node2',
                                        itemConfigs:[
                                            {html:'node1'},
                                            {html:'node2'}
                                        ]
                                    }
                                ],
                                listeners:{
                                }
                            }
                        ],
                        listeners:{}
                    }
                ],
                listeners:{
                    click:function(e){
                        console.log('item11'+e.text);
                    }
                }
            },
            '-',
            {
                html:'item12',
                listeners:{
                    click:function(e){
                        console.log(e.text);
                    }
                }
            }
        ],
        listeners : {
            click : function(e) {
                alert(e.text);
            }
        }
    }));
    menuBar.addComponent(new Sui.menu.Menu({
        html:'菜单2',
        itemConfigs:[
            {html:'菜单0_1'},
            {html:'菜单0_2'},
            {html:'菜单0_3',
                itemConfigs:[
                    { html:'node1'},
                    {
                        html:'node2',
                        itemConfigs:[
                            {html:'node1'},
                            {html:'node2'}
                        ]
                    }
                ]
            }
        ]
    }));
