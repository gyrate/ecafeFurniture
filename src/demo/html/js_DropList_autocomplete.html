Example:使用Droplist 级联功能

var select = new Sui.form.DropList({
    applyTo : 'input',
    name : 'continents',
    //value:'Asia',
    searchable : true,
    comboMaxHeight:150,
    hideIconElement:true,//隐藏下拉选择层触发按钮
    forceSelect :false, //允许手动输入任何值
    items : [
        { label : '亚洲', value : 'Asia'},
        { label : '美洲',value : 'Americas'},
        { label : '欧洲',value : 'Europe'},
        { label : '大洋洲',value : 'Oceania'},
        { label : '非洲',value : 'Africa'}
    ],
    listeners:{
        selected : function(event) {
        }
    }
});

select.getLayer().addExcludeElementsClick($('#trigger'));
$('#trigger').click(function() {
    select.toggleExpand();
});
