    菜单组件

    Sui.menu.Menu(${option})

    menu.js

    类

    Sui.TagComponent

    无

    Sui.menu.Menu

    jquery-1.7.1.js , sui.js ,suibase.js , menu.js

    sui.menu.css

    参数

    ${option}所能够包含的属性。*表示该属性为必填项

    参数名(类型) 实例 说明
    renderTo(String) 'div'

    渲染到组件的id
    html(String) '菜单项A'

    菜单的描述值,默认为''
    itemConfigs(Array) [ {html:'item1',icon:'icon.png',Listeners:{...}},{html:'item2',icon:'icon.png',Listeners:{...}},.. ]

    菜单项配置参数
    layerClass(String) 'setting_layer'

    菜单主体的外部样式，默认为null
    hideOnClick(Boolean) true (默认)

    点击每个可执行菜单项后是否关闭整个菜单

    公共方法

    方法
    查看所有方法： Sui.menu.Menu.html

    例子

    Example:创建一个标准Menu组件

    折叠|展开 
    menu = new Sui.menu.Menu({
    html:'菜单测试',
    applyTo:'menu',
    itemConfigs:[
        {
            html:'菜单项A',
            action:function(obj) {
                alert('exclude in Action ');
            },
            listeners:{
                click:function() {
                    alert('exclude in Listeners ');
                }
            }
        },
        {
            html:'菜单项B',
            itemConfigs:[
                {html:'菜单项A1',
                 defaultClass:'sui_menuitem sui_menuitem_test'//该菜单项添加了特殊样式
                },
                {html:'菜单项A2',
                 iconBaseUrl:'../../images/table/',
                 icon:'checked.png',
                },
                {html:'菜单项A3',
                    itemConfigs:[
                        {html:'菜单项A31'},
                        {html:'菜单项A32'},
                        {html:'菜单项A33'}
                    ]
                }
            ]
        },
        { html:'菜单项C',action:function(obj){console.log(obj);} }
    ]
})

    Example:显示、隐藏菜单

    折叠|展开 
    menu.toggle();
