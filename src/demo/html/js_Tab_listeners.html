Example:点击tab标签触发事件异步更新内容

var tab = new Sui.Tab({
    applyTo:'switch' ,
    menuClass:'menu_item',
    contentClass:'content_item',
    trigger:'click',
    needOperator:true,
    //currentIndex:2,
    listeners:{
        changeTab:function(e) {

            var index = e.index ;
            tab.setContentItemDom( tab.getContentItemByIndex(index) ,{
                needIframe:true,
                url:'js_Tab_listeners_test.html'
            });
        }
    }
}) ;
